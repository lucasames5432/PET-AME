packages = ["matplotlib", "numpy"] import matplotlib.pyplot as plt import numpy as np import math import random from pyscript import display, window from js import document # Variável global para armazenar o exercício atual current_ex = {} def criar_figura(m, n): h = math.sqrt(m * n) fig, ax = plt.subplots(figsize=(8, 4)) # Coordenadas: A (topo), B (esquerda), C (direita) # Colocamos o pé da altura no (0,0) A = [0, h] B = [-n, 0] C = [m, 0] # Desenha o triângulo ax.plot([A[0], B[0], C[0], A[0]], [A[1], B[1], C[1], A[1]], color='#38703b', lw=3) # Desenha a altura tracejada ax.plot([0, 0], [0, h], color='red', linestyle='--', lw=2, label=f'h') # Ajustes visuais para combinar com o site ax.set_aspect('equal') ax.axis('off') ax.set_title("Triângulo Retângulo", color='#37474f', fontfamily='sans-serif') return fig def gerar_novo_exercicio(event=None): global current_ex m = random.randint(2, 9) n = random.randint(2, 9) h = math.sqrt(m * n) current_ex = { 'pergunta': f"As projeções dos catetos sobre a hipotenusa medem m = {m} cm e n = {n} cm.
